			\item[-] service = abstrait
		\end{itemize}
			
	\subsection{L'entreprise, une organisation focalisée sur la fonction commerciale}
	
	\subsubsection{Fonction commerciale: définir ce que l'on entend par marketing; par besoin}

		\begin{itemize}
			\item[-] Promotion du produit
			\item[-] Ensemble de méthodes et moyens dont dispose une organisation pour promouvoir, dans les publics auxquels elle s'intéresse,
				des comportements favorables à la réalisation de ces propres objectifs
			\item[-] B2C = business to customers
			\item[-] B2B = business to business (besoin lié à l'activité de l'entreprise)
		\end{itemize}
			
	\subsubsection{Mettre en évidence la fonction prioritaire au sens économique}
			
			\begin{itemize}
				\item[-] Marketing, car l'entreprise cherche à vendre ses produits
